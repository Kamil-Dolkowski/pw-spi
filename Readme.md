# dodanie usera i maila
git config --local user.name ""
git config --local user.email ""

# nowe repo
git init

# klonowanie repo
git clone
# tworzy nowy katalog 
git close https://

# klon do aktualnego katalogu 
git close https:// .

git close https:// /nowy-katalog

# dodanie pliku do staging
git add nazwa-pliku
git add *   (wszystkie)
git add .   (aktualny katalog)

# komitowanie zmian
git commit -m"zmiany"

# status zmian
git status

# pobranie wszystkich zmian z repo
git fetch

# pobierz zmiany
git pull

# wypchnij zmiany
